version: '3'
services:
   mysql:
    platform: linux/x86_64
    image: "mysql:5.7"
    environment:
      MYSQL_ROOT_PASSWORD: root
      MYSQL_DATABASE: "${DB_NAME}"
      MYSQL_USER:     "${DB_USER}"
      MYSQL_PASSWORD: "${DB_PASS}"
    ports:
      - "${DB_PORT}:3306"
    tty: true
    volumes:
      - "./db:/var/lib/mysql:cached"
    restart: always
